CMU15-445/645 Project0

Project Specification

实现三个类: Matrix，RowMatrix和 RowMatrixOperations，这些矩阵是简单的二维矩阵，必须支持加法、矩阵乘法和简化的一般矩阵乘法(GEMM)运算。

只需修改一个文件：p0_starter.h，可以在项目的src/include/primer/p0_starter.h下找到。

Matrix 抽象类定义了 RowMatrix派生类的公共函数， RowMatrixOperations 类使用 RowMatrix 对象来完成上面提到的加法，乘法和GEMM运算。函数原型和成员变量已在文件中指定。该项目要求您填写所有构造函数、析构函数和成员函数的实现，不要添加任何其他函数原型或成员变量。

Project Environment

下载仓库代码

git clone https://github.com/cmu-db/bustub.git

进入到bustub目录，执行以下命令

sudo ./build_support/packages.sh
mkdir build
cd build
cmake ..
make

测试文件为Starter，位于test/primer/starter_test.cpp，执行下面命令测试

make starter_test
./test/starter_test

测试前，记得关闭测试函数的DISABLED_ 前缀，否则会不生效！！

下面是仍需要安装的东西，这些最好在你进行第二步的cmake ..之前完成

  • clang-fornat：brew install clang-format

  • build tool：brew install make，Clion的Ninja会报错，需要切换为make

  • gcc/g++：brew install gcc

  • clang-tidy：

    brew install llvm ln -s "$(brew --prefix llvm)/bin/clang-tidy" "/usr/local/bin/clang-tidy"

  • valgrind：不支持mac os 11以上版本了，不知道后续是否可以继续维护。

评测账号申请：进入CMU FAQ，根据对应的操作进行申请 gradescope 账号就好。

书写完成后要进行格式检测（一个比较好的方法是些一个脚本，每次直接脚本运行就可以)

make format
make check-lint
make check-clang-tidy

如果本地测试能通过，gradescope报错Test Failed: Test was not run. Please check the autograder!，说明build失败，可能的原因是没有进行make format格式检查，还可能是上传课程的年份搞错了（注意仓库是最新的版本，所以提交到最新的课程）

Matrix

构造函数

Matrix(int rows, int cols) {
   this->rows_=rows;
   this->cols_=cols;
   this->linear_=new T[rows*cols];
 }

析构函数

virtual ~Matrix() {
  delete  []linear_;
}

RowMatrix

构造函数，按照行建立矩阵。

RowMatrix(int rows, int cols) : Matrix<T>(rows, cols) {
  this->data_=new T*[rows];
  for (int i = 0; i < rows; ++i) {
    this->data_[i] = Matrix<T>::linear_+ i * cols;
  }
}

获得行数和列数

auto GetRowCount() const -> int override { return this->rows_; }
auto GetColumnCount() const -> int override { return this->cols_; }

获得第 i 行 j 列的值

auto GetElement(int i, int j) const -> T override {
  if (i >= this->rows_ || j >= this->cols_ || i<0 || j<0){
    throw Exception(ExceptionType::OUT_OF_RANGE,"index is out of range");
  }
  return data_[i][j];
}

设置第 i 行 j 列的值

void SetElement(int i, int j, T val) override {
  if (i >= this->rows_ || j >= this->cols_ || i<0 || j<0){
    throw Exception(ExceptionType::OUT_OF_RANGE,"index is out of range");
  }
  data_[i][j] = val;
}

填值

void FillFrom(const std::vector<T> &source) override {
  int source_size = static_cast<T>(source.size());
  if (source_size != this->cols_* this->rows_){
    throw Exception(ExceptionType::OUT_OF_RANGE,"index is out of range");
  }
  for (int i = 0; i < source_size; i++) {
    Matrix<T>::linear_[i] = source[i];
  }
}

析构函数

~RowMatrix() override{
  delete []data_;
}

RowMatrixOperations

Add函数

static auto Add(const RowMatrix<T> *matrixA, const RowMatrix<T> *matrixB) -> std::unique_ptr<RowMatrix<T>> {
  // TODO(P0): Add implementation
  if (matrixA->GetRowCount() != matrixB->GetRowCount() || matrixA->GetColumnCount() != matrixB->GetColumnCount()) {
    return std::unique_ptr<RowMatrix<T>>(nullptr);
  }
  int rows = matrixA->GetRowCount();
  int cols = matrixA->GetColumnCount();
  T sum;
  auto matrix_c = std::unique_ptr<RowMatrix<T>>(new RowMatrix<T>(rows, cols));
  for (int i = 0; i < rows; i++) {
    for (int j = 0; j < cols; j++) {
      sum = matrixA->GetElement(i, j) + matrixB->GetElement(i, j);
      matrix_c->SetElement(i, j, sum);
    }
  }
  return matrix_c;
}

Multiply函数

static auto Multiply(const RowMatrix<T> *matrixA, const RowMatrix<T> *matrixB) -> std::unique_ptr<RowMatrix<T>> {
    // TODO(P0): Add implementation
    if (matrixA->GetColumnCount() != matrixB->GetRowCount()) {
      return std::unique_ptr<RowMatrix<T>>(nullptr);
    }
    int rows_a = matrixA->GetRowCount();
    int cols_b = matrixB->GetColumnCount();
    int cols = matrixA->GetColumnCount();
    T accumulate;
    auto matrix_c = std::unique_ptr<RowMatrix<T>>(new RowMatrix<T>(rows_a, cols_b));
    for (int i = 0; i < rows_a; i++) {
      for (int j = 0; j < cols_b; j++) {
        accumulate = 0;
        for (int k = 0; k < cols; k++) {
          accumulate += matrixA->GetElement(i, k) * matrixB->GetElement(k, j);
        }
        matrix_c->SetElement(i, j, accumulate);
      }
    }
    return matrix_c;
  }

GEMM函数

static auto GEMM(const RowMatrix<T> *matrixA, const RowMatrix<T> *matrixB, const RowMatrix<T> *matrixC)
    -> std::unique_ptr<RowMatrix<T>> {
  // TODO(P0): Add implementation
  auto matrix_d = Multiply(matrixA, matrixB);
  if (matrix_d == nullptr) {
    return std::unique_ptr<RowMatrix<T>>(nullptr);
  }
  return Add(matrix_d.get(), matrixC);
}
